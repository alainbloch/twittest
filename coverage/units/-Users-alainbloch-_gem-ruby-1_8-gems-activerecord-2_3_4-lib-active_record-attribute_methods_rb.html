342     def respond_to?(method, include_private_methods = false)
 
343       method_name = method.to_s
 
344       if super
 
345         return true
 
346       elsif !include_private_methods && super(method, true)
 
347         # If we're here than we haven't found among non-private methods
 
348         # but found among all methods. Which means that given method is private.
 
349         return false
 
350       elsif !self.class.generated_methods?
 
351         self.class.define_attribute_methods
 
352         if self.class.generated_methods.include?(method_name)
 
353           return true
 
354         end
 
355       end
 
356         
357       if @attributes.nil?
 
358         return super
 
359       elsif @attributes.include?(method_name)
 
360         return true
 
361       elsif md = self.class.match_attribute_method?(method_name)
 
362         return true if @attributes.include?(md.pre_match)
 
363       end
